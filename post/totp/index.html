2017-11-15

看过阮一峰的这篇日志，写一个Golang的实现。

Base32实现

对于秘钥的发布，生成可以使用Base32进行编码，如果秘钥太长可以生成二维码扫码认证。Base32的实现可以参考 Base64 in Golang
下面的实现没有使用Base32。

TOTP

TOTP Stands for Time-Based One-Time Password
由秘钥和时间作为参数生成一个Password，秘钥在客户端(用户)和服务端各自一份，用户生成这个Password后进行登录，服务端用相同的秘钥进行生成Password，再比对密码。
超出时间(比如30s)，则Password不相同，Password只在规定的时间内有效，所以服务器的时间和客户端的时间需要是同步的。

代码如下：

    package main

    import (
        "crypto/hmac"
        "crypto/sha1"
        "fmt"
        "os"
        "time"
    )

    /**
     *  数字 => Byte数组
     */
    func toByte(input int64) []byte {
        var output []byte
        mask := int64(0xFF)
        shifts := [8]uint16{56, 48, 40, 32, 24, 16, 8, 0}
        for _, shift := range shifts {
            tmp := byte((input >> shift) & mask)
            output = append(output, tmp)
        }
        return output
    }

    func toDigit(input []byte) uint32 {
        return (uint32(input[0]) << 24) + (uint32(input[1]) << 16) + (uint32(input[2]) << 8) + uint32(input[3])
    }

    /**
     *  生成器
     */
    func oneTimePassword(secret, ts []byte) uint32 {
        //hmac-sha1生成Hash
        hmacSha1 := hmac.New(sha1.New, secret)
        hmacSha1.Write(ts)
        //生成的Hash是20Byte的数据，相当于40个十六进制数
        hash := hmacSha1.Sum(nil)

        //缩短Hash，取Hash slice的最后一个字节数字&15
        offset := hash[len(hash)-1] & 0x0F
        //极限情况就是[15:19]，不会超过20字节
        subhash := hash[offset : offset+4]

        //忽略最高位，以防溢出
        subhash[0] = subhash[0] & 0x7F

        //32bit转换为数字
        digit := toDigit(subhash)

        digit = digit % 1000000

        return digit
    }

    func main() {
        if len(os.Args) < 2 {
            fmt.Fprint(os.Stderr, "Missing Parameter")
            os.Exit(1)
        }

        //这里没有选择Base32decode
        secret := os.Args[1]
        secretByte := []byte(secret)
        ts := time.Now().Unix()

        pwd := oneTimePassword(secretByte, toByte(ts/30))

        remain := 30 - ts%30
        fmt.Println(pwd, " | 可用时间", remain, "s")
    }
