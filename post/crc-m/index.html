循环冗余检验

2017-11-20

学计算机网络的时候接触过，CRC对于报文的完整性检验，记录下。

首先计算待传输数据的冗余码(余数)，将余数加到数据后面进行传输，另一端收到后将数据对同一个除数进行取余，若余数为0，则数据完整。

基础知识

  • LSB
    最低有效位(Least Significant Bit)
  • MSB
    最高有效位(Most Significant Bit)
  • 生成多项式
    生成多项式的选择是CRC算法实现中最重要的部分，所选择的多项式必须有最大的错误检测能力，同时保证总体的碰撞概率最小。多项式最重要的属性是它的长度，也就是最高非零系数的数值，因为它直接影响着计算的校验和的长度。比如CRC-8 = x^8+x^2+x+1，即 10000 0111，去掉MSB即 0000 0111(0x07)。

  • 反转字节

    func ReverseByte(val byte) byte {
    var rval byte = 0
    for i := uint8(0); i < 8; i++ {
        if val&(1<<i) != 0 {
            rval |= 0x80 >> i
        }
    }
    return rval
}

  • CRC-M表格
    8位，一共256种可以提前生成一个对应的CRC表格。

    func makeTable(params Params) *Table {
    table := new(Table)
    table.params = params

    for i := 0; i < 256; i++ {
        num := uint8(i)
        //遍历整个字节的每一位
        for j := 0; j < 8; j++ {
            //MSB是否为1，为1则与多项式异或
            yep := (num & 0x80) != 0
            num <<= 1
            if yep {
                num ^= params.Poly
            }
        }
        table.data[i] = num
    }
    return table
}

思路

A = 0011 1110，C1=00110000B，C2=1110B
A = C1 XOR C2
B = Poly(多项式)
则 A XOR B = （C1 XOR C2） XOR B =(C1 XOR B )XOR C2
这里的例子是CRC-4，下面的Update函数是CRC-8。

完整代码

    package main

    import (
        "fmt"
    )

    /**
     *  Poly 多项式
     *  Init 初始化值
     *  RefIn 如果此值为true，表示输入的数据需要进行比特翻转，也就是Bit7要变成最低位， Bit0要变成最高位
     *  RefOut 如果此值为true，这表示进行异或后算出来的CRC需要进行整个比特翻转
     *  XorOut 最终异或的值
     */
    type Params struct {
        Poly   uint8
        Init   uint8
        RefIn  bool
        RefOut bool
        XorOut uint8
    }

    type Table struct {
        params Params
        data   [256]uint8
    }

    func makeTable(params Params) *Table {
        table := new(Table)
        table.params = params

        for i := 0; i < 256; i++ {
            num := uint8(i)
            //遍历整个字节的每一位
            for j := 0; j < 8; j++ {
                //MSB是否为1，为1则与多项式异或
                yep := (num & 0x80) != 0
                num <<= 1
                if yep {
                    num ^= params.Poly
                }
            }
            table.data[i] = num
        }
        return table
    }

    /**
     *  反转字节
     */
    func ReverseByte(val byte) byte {
        var rval byte = 0
        for i := uint8(0); i < 8; i++ {
            if val&(1<<i) != 0 {
                rval |= 0x80 >> i
            }
        }
        return rval
    }

    func ReverseUint8(input uint8) uint8 {
        return ReverseByte(input)
    }

    func Update(init uint8, data []byte, table *Table) uint8 {
        for _, value := range data {
            if table.params.RefIn {
                value = ReverseByte(value)
            }
            init = table.data[init^value]
        }
        return init
    }

    func Complete(crc uint8, table *Table) uint8 {
        if table.params.RefOut {
            crc = ReverseUint8(crc)
        }
        return crc ^ table.params.XorOut
    }

    func main() {
        crc8 := Params{0x07, 0x00, false, false, 0x00}
        table := makeTable(crc8)
        init := table.params.Init
        crc := Update(init, []byte{0x12, 0x55}, table)
        crc = Complete(crc, table)

        fmt.Println(crc)
    }
