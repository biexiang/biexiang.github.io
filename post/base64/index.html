Base64 in Golang

2017-11-15

依据码表对字符进行重新编码，可逆。

Ascii字符都是8位，范围在256以内。而Base64是使用6位表示一个字符所以范围是64以内，Base32是使用5位表示一个字符，范围在32以内，下面是码表字符，索引为数值。

    BASE64 string = "ABCDEFGHIJKLMNOPQRSTUVWZYZabcdefghijklmnopqrstuvwxyz0123456789+/"
    BASE32 string = "ABCDEFGHIJKLMNOPQRSTUVWZYZ234567"

上面Base64编码如果结果不是4的整数倍需要补”=“号，少几个补几个。Base32也是对8进行取余Pad。

编码的过程就是遍历字符串，去每一个字符的Ascii码表的十进制值，转换为二进制，不满8位，高位补0，拼接整个字符串的二进制，以6个切分取编码对应索引的值组成新的字符串，最后判断补”=“。
解码是编码的逆过程。

    package main

    import (
        "fmt"
        "os"
        "strconv"
        "strings"
    )

    var (
        BASE64 string = "ABCDEFGHIJKLMNOPQRSTUVWZYZabcdefghijklmnopqrstuvwxyz0123456789+/"
        BASE32 string = "ABCDEFGHIJKLMNOPQRSTUVWZYZ234567"
    )

    /**
     * 填充0
     */
    func strpad(target string, direction int, length int, pad string) string {
        if len(target) > length {
            fmt.Fprintln(os.Stderr, "no need to pad")
            os.Exit(1)
        }

        if direction == 1 {
            //left
            for {
                if len(target) >= length {
                    return target
                }
                target = pad + target
            }
        } else if direction == 0 {
            //right
            for {
                if len(target) >= length {
                    return target
                }
                target = target + pad
            }
        } else {
            fmt.Fprintln(os.Stderr, "wrong direction")
            os.Exit(1)
        }

        return "error"
    }

    /**
     * 返回数字索引
     */
    func getIndex(str string, word byte) int {
        for i := 0; i < len(str); i++ {
            if string(word) == "=" {
                return 0
            }

            if str[i] == word {
                return i
            }
        }
        return -1
    }

    /**
     * str => dec => bin => process
     */
    func enBase64(input string) string {
        var (
            binstr string = ""
            output string = ""
        )

        for i := 0; i < len(input); i++ {
            dec := int64(input[i])
            bin := strconv.FormatInt(dec, 2)
            strbin := strpad(string(bin), 1, 8, "0")
            binstr += strbin
        }

        // 6*4 = 8*3 填充左边的0
        num := (6 - len(binstr)%6) + len(binstr)
        final := strpad(binstr, 0, num, "0")

        for i := 0; i < num; i += 6 {
            bin := final[i : i+6]
            dec, err := strconv.ParseInt(bin, 2, 64)
            if err != nil {
                fmt.Fprintln(os.Stderr, err)
                os.Exit(1)
            }

            output += string(BASE64[dec])
        }

        //添加等于号
        num = (4 - len(output)%4) + len(output)
        output = strpad(output, 0, num, "=")

        return output
    }

    /**
     *  str => dec => bin => process
     */
    func deBase64(input string) string {
        var (
            binstr string = ""
            output string = ""
        )

        //删除等于号
        input = strings.TrimRight(input, "=")

        for i := 0; i < len(input); i++ {
            index := getIndex(BASE64, input[i])
            if index == -1 {
                fmt.Fprintln(os.Stderr, "Wrong decode string in ", i)
                os.Exit(1)
            }
            bin := strconv.FormatInt(int64(index), 2)
            strbin := strpad(string(bin), 1, 6, "0")
            binstr += string(strbin)
        }

        length := len(binstr)
        overflow := length % 8
        binstr = binstr[0 : length-overflow]

        for i := 0; i < len(binstr); i += 8 {
            dec, err := strconv.ParseInt(binstr[i:i+8], 2, 64)
            if err != nil {
                fmt.Fprintln(os.Stderr, err)
                os.Exit(1)
            }

            output += string(dec)
        }
        return output
    }

    /**
     *  str => dec => bin => process
     */
    func enBase32(input string) string {
        var (
            binstr string = ""
            output string = ""
        )

        for i := 0; i < len(input); i++ {
            dec := int64(input[i])
            bin := strconv.FormatInt(dec, 2)
            strbin := strpad(string(bin), 1, 8, "0")
            binstr += strbin
        }

        // 5*8 = 8*5 填充左边的0
        num := (5 - len(binstr)%5) + len(binstr)
        final := strpad(binstr, 0, num, "0")

        for i := 0; i < num; i += 5 {
            bin := final[i : i+5]
            dec, err := strconv.ParseInt(bin, 2, 64)
            if err != nil {
                fmt.Fprintln(os.Stderr, err)
                os.Exit(1)
            }

            output += string(BASE32[dec])
        }

        //添加等于号
        num = (8 - len(output)%8) + len(output)
        output = strpad(output, 0, num, "=")

        return output

    }

    func deBase32(input string) string {
        var (
            binstr string = ""
            output string = ""
        )

        //删除等于号
        input = strings.TrimRight(input, "=")

        for i := 0; i < len(input); i++ {
            index := getIndex(BASE32, input[i])
            if index == -1 {
                fmt.Fprintln(os.Stderr, "Wrong decode string in ", i)
                os.Exit(1)
            }
            bin := strconv.FormatInt(int64(index), 2)
            strbin := strpad(string(bin), 1, 5, "0")
            binstr += string(strbin)
        }

        length := len(binstr)
        overflow := length % 8
        binstr = binstr[0 : length-overflow]

        for i := 0; i < len(binstr); i += 8 {
            dec, err := strconv.ParseInt(binstr[i:i+8], 2, 64)
            if err != nil {
                fmt.Fprintln(os.Stderr, err)
                os.Exit(1)
            }

            output += string(dec)
        }
        return output
    }

    func main() {
        str := "liang"
        tst := enBase64(str)
        fmt.Println(tst)
        fmt.Println(deBase64(tst))

        tst = enBase32(str)
        fmt.Println(tst)
        fmt.Println(deBase32(tst))
    }
