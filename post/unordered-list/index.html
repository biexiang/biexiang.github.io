无序链表排序

2017-10-14

Use Golang，定义一个链表的结构，随机产生链表，打印链表。

  • 定义链表结构

    //定义结构
type Node struct {
    Data int
    Next *Node
}

  • 给定长度，随机产生无序链表

    //创建一个指定长度的无序的链表
func Create(size int) *Node {
    var root *Node = &Node{
        Data: 64,
        Next: nil,
    }

    head := root
    for i := 0; i < size; i++ {
        tmp := &Node{
            Data: rand.Intn(1000),
            Next: nil,
        }
        head.Next = tmp
        head = head.Next
    }

    return root
}

  • 归并排序

    //龟兔赛跑
func (root *Node) getMiddle() *Node {
    head1 := root
    head2 := root

    for head2.Next != nil && head2.Next.Next != nil {
        head2 = head2.Next.Next
        head1 = head1.Next
    }

    return head1
}

//归并 切
func (root *Node) MergeSort() *Node {
    //获取链表的中点，切分

    if root.Next == nil {
        return root
    }

    middle := root.getMiddle()
    head1 := root
    head2 := middle.Next
    middle.Next = nil

    //递归切
    head1 = head1.MergeSort()
    head2 = head2.MergeSort()

    res := _mergeSort(head1, head2)
    return res
}

//归并 排  从小到大
func _mergeSort(head1, head2 *Node) *Node {
    //不判断
    if head1 == nil {
        return head2
    }
    if head2 == nil {
        return head1
    }

    var sorted *Node
    if head1.Data < head2.Data {
        head1.Next = _mergeSort(head1.Next, head2)
        sorted = head1
    } else {
        head2.Next = _mergeSort(head1, head2.Next)
        sorted = head2
    }

    return sorted
}

  • 打印

    //打印链表
func (root *Node) print() {
    for root != nil {
        fmt.Println(root.Data)
        root = root.Next
    }
}

  • 总结： 排序的时候采用归并，速度不同找到中间节点，先拆后排。时间复杂度为nlogn。
