  • 控制住并发数，并发为3，如果大于则Enter时管道会阻塞，直到有Leave的情况。

    var limit = make(chan int, 3)

func main() {
    for _, w := range work {
        go func() {
            limit <- 1
            w()
            <-limit
        }()
    }
    select{}
}

  • 带缓存的ch进行并发控制、状态监控，通过for switch循环进行状态监控，在每一个状态输出不一样的东西。

    type struct Vch {
    CH chan bool 
}

func NewVch() Vch {
    return &Vch{
        CH: make(chan bool,5),
    }
}

func (vch Vch) Len() int {
    return len(vch.CH)
}
func (vch Vch) Cap() int {
    return cap(vch.CH)
}
func (vch Vch) Idle() bool {
    return len(vch.CH) == 0
}
func (vch Vch) Busy() bool {
    return len(vch.CH) == cap(vch.CH)
}
